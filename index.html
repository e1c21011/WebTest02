window.startVR = function() { console.log("VR Demo starting..."); document.getElementById("vrButton").style.display = 'none'; // ボタンを非表示にする const viewerOptions = { 'initialCameraLookAt': [0.20786, -0.68154, -0.27311], 'webXRMode': null // ARがサポートされない場合は通常モードで表示 }; if (navigator.xr && navigator.xr.isSessionSupported) { navigator.xr.isSessionSupported('immersive-ar').then((supported) => { if (supported) { viewerOptions.webXRMode = GaussianSplats3D.WebXRMode.AR; console.log("AR mode supported, enabling AR."); } else { console.log("AR mode not supported, proceeding with regular viewer."); } initializeViewer(viewerOptions); }); } else { console.log("WebXR not supported, using regular viewer."); initializeViewer(viewerOptions); } } function initializeViewer(options) { const viewer = new GaussianSplats3D.Viewer(options); let path = 'assets/data/converted_file.ksplat'; console.log("Loading scene from: ", path); viewer.addSplatScene(path, { 'rotation': new THREE.Quaternion().setFromUnitVectors(new THREE.Vector3(0.01933, -0.75830, -0.65161).normalize(), new THREE.Vector3(0, 1, 0)).toArray(), 'scale': [0.25, 0.25, 0.25], 'position': [0, 0.5, 0] }) .then(() => { console.log("Scene loaded successfully, starting viewer..."); viewer.start(); }) .catch((error) => { console.error("Error loading the scene:", error); }); }
